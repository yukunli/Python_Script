function submitrequest(strAction){
	var elementform = document.getElementById("form1");
	var requestaction = $("#requestaction");
	if(strAction == "re build run" || strAction == "re run" || strAction == "clone" || strAction == "clone fail case" ){ 
		var j = 0;
		$("input[name='requestdetailid[]']").each(function(i){
			if($(this).prop("checked") == true){
				j++;
				return false;
			}
		})
		if(j == 0){
			if (strAction != "clone" && strAction != "clone fail case"){
				alert("Please Select At Lease One Records!");
				return;
			}
		}
	}
	switch(strAction){
		case "addcase":
			elementform.action="/dapeng/addMcuautoRequest2/";
			elementform.submit();
			break;
		case "makeperiodic":
			elementform.action="/dapeng/EditMcuautoPeriodicRequest/0/";
			elementform.submit();
			break;
		case "editrequest":
			elementform.action="/dapeng/EditMcuautoRequest/0/";
			elementform.submit();
			break;
		case "removecase":
			var ret = confirm("Are you sure to remove cases from this request?");
			if (ret == true)
			{
                requestaction.val(strAction);
                elementform.submit();
			}
			break;
		case "stop":
			var ret = confirm("Are you sure to stop this request?");
			if (ret == true)
			{
                elementrebuildruns = document.getElementsByName("rebuildrun");
                window.console.log(elementrebuildruns[0]);
                requestaction.val(strAction);
                elementform.submit();
			}
			break;
		case "re build run":
			var ret = confirm("Are you sure to re-build and re-run these test cases?");
			if (ret==true)
			{
                requestaction.val(strAction);
                elementform.submit();
			}
			break;
		case "re run":
			var ret = confirm("Are you sure to re-run without re-build these test cases?");
			if (ret == true)
			{
                requestaction.val(strAction);
                elementform.submit();
			}
			break;
		case "report":
			var dialogParams = {
				height:150,
				width: 400,
				modal: true,
				autoOpen: false,
				close: function(){$(this).remove();},
				buttons:{
					Ok:function(){
						requestaction.val(strAction);
						$("#detailjiraid").val($("#jira_report").val());
						elementform.submit();
					}, 
					Cancel:function(){
						$(this).dialog('close');
					}
				}
			}; 
			var htmlParams = "JIRA ID: <input id='jira_report' value=''>"
			myDialog = $('<div id="report_dialog"></div>').html(htmlParams).dialog(dialogParams);
			myDialog.dialog('open');
			break;
		case "del report":
			var ret = confirm("Are you sure to delete these reported?");
			if (ret == true)
			{
                requestaction.val(strAction);
                elementform.submit();
			}
			break;
		default:
			requestaction.val(strAction);
			elementform.submit();
			break;
	}
}

function openShutManager(src1, src2, target){   
	
	var	strSrc1 = $("#" + src1).text();
	var	strSrc2 = $("#" + src2).text();
//console.debug(strSrc1);
//console.debug(strSrc2);
//console.debug(target);
	var objTarget = typeof target == "string" ? $("#" + target) : $(target);
//console.debug(objTarget.html());
	if(objTarget.html() == "all")
		return;
	if(objTarget.html() != strSrc1){
		objTarget.html(strSrc1); 
		return;
	}
	if(objTarget.html() != strSrc2){
		objTarget.html(strSrc2); 
		return;
	}		
}

function submitReported(src){
    src = $(src);
	var ids = src.attr("id").split("isreported_");	
	var srcid = "jiraid_" + ids[1];
	var value = 0;
	if(src.prop("checked") == true && src.val() != ""){
		value = 1;
		$('#' + srcid).html("<input id='" + ids[1] + "' value='' onchange='submitJiraid(this, " + value + ")'>");
	}
	else{
		submitJiraid('#' + ids[1], value);
	}
}
function submitJiraid(src, value){
	src = $(src);
	id = src.attr("id");
	srcValue = src.val();
    var isReportedId = "isreported_" + id
    var isReportedNode = $('#' + isReportedId)
    var isReParentNode = isReportedNode.parent();
    var parentNode = $("#" + id).parent();
    var csrfmiddlewaretoken = document.getElementsByName("csrfmiddlewaretoken")[0].value
    var postData = {requestdetailid:id, isreported:value, jiraid:srcValue, csrfmiddlewaretoken:csrfmiddlewaretoken};	
    $.post("/dapeng/detailReported/", postData, function(data, status){
        if(data == 1){
            isReParentNode.html("<span id='" + isReportedId+ "' onclick='submitReported(this)'>Reported&nbsp&nbsp<img src='/dapeng/static/pics/delete.png'></span>");
            var srcValues = srcValue.split(",");
            var ret = "";
            var length = srcValues.length
            $.each(srcValues, function(i, value){
                
                ret += '<a id="' + id + '" href="http://sw-jira.freescale.net/browse/' + value + '"  target="_blank">' + value
                if(i == length-1)
                    ret += '</a>';
                else
                    ret += ',</a>';
            });
            parentNode.html(ret);
        }
        else if(data == 0){
            isReParentNode.html("<input type='checkbox' name='isreported[]' id='" + isReportedId + "' value='off' onclick='submitReported(this)'>");
            parentNode.html("");
            $('#' + isReportedNode.id).attr("checked", false);
        }
        else{
            alert("fail");
        }
    });
}
